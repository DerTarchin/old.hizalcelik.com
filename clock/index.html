void setup() {
  size(1280, 720);
  for (int c=0; c<8; c++) {
    for (int r=0; r<3; r++) {
      int index = c+(8*r);
      cX[index]=(width/2)-(radius*4-radius/2)+(radius*c);
      cY[index]=(height/2)-radius+(radius*r);
    }
  }

  setTime();
  millisRolloverTime=0;
}

void draw() {
  background(#ffffff);
  fill(0);
  rect(width/2-radius/2, height/2-120, radius, radius*2);
  fill(#ffffff);
  noStroke();
  ellipse(width/2, height/2-radius/2, radius/2.68, radius/2.68);
  ellipse(width/2, height/2+radius/2, radius/2.68, radius/2.68);

  for (int i=0; i<24; i++) {
    stroke(0);
    strokeWeight(4); 
    fill(#ffffff);
    ellipse(cX[i], cY[i], radius, radius);
    strokeWeight(8);

    showTimeMove(i);
    //showTimeTogether(i);

    if (minute()+1==60) {
      println("Next min: 00");
    } else {
      println("Next min: "+(minute()+1));
    }
    if (hour()+1==24) {
      println("Next hr: 00");
    } else {
      println("Next hr: "+(hour()+1));
    }
  }
}

float backwards(float num) {
  float newnum = num;
  if (num==0) { 
    newnum= 0;
  }
  if (num==90) { 
    newnum= 270;
  }
  if (num==135) { 
    newnum= 225;
  }
  if (num==180) { 
    newnum= 180;
  }
  if (num==270) { 
    newnum= 90;
  }
  return newnum;
}

    About the Project

    For this class assignment, we had to make a unique way to display time. In my application, the clocks are designed to work in unison to display the time in a digital style. The animation cycle occurs twice a minute, each lasting 15 seconds. The time displayed is the same before and after the first animation cycle, but the second cycle ends on the next minute. This application works real-time using the system timestamp. This was written in Processing.
